Phase 3Utils — pdf_extractor, jd_scraper, resume_generator
1
utils/pdf_extractor.py
↓
resume-agent/utils/pdf_extractor.py
Extracts text from PDF using pdfplumber and from DOCX using python-docx. Auto-detects file type by extension.
import pdfplumber, docx, io def extract_resume_text(uploaded_file) -> str: filename = uploaded_file.name.lower() if filename.endswith(".pdf"): with pdfplumber.open(uploaded_file) as pdf: return "\n".join( p.extract_text() for p in pdf.pages if p.extract_text() ).strip() elif filename.endswith(".docx"): doc = docx.Document(io.BytesIO(uploaded_file.read())) return "\n".join( p.text for p in doc.paragraphs if p.text.strip() ).strip()
python -c "from utils.pdf_extractor import extract_resume_text; print('✅ pdf_extractor ok')"

Phase 7LangGraph Pipeline
1
Create graph/pipeline.py — ResumeState TypedDict
↓
resume-agent/graph/pipeline.py
The shared state dict that flows through all 4 agents. Each agent reads from it and adds its output.
from typing import TypedDict, Optional from langgraph.graph import StateGraph, END class ResumeState(TypedDict): resume_text: str jd_text: str include_suggested_projects: bool parsed_jd: Optional[dict] # Agent 1 fills this evaluation: Optional[dict] # Agent 2 fills this gap_analysis: Optional[dict] # Agent 3 fills this final_resume: Optional[str] # Agent 4 fills this current_step: Optional[str] error: Optional[str]
Teaching moment: The {**state, "parsed_jd": parsed} pattern spreads all existing keys and adds only the new one. This is how state grows richer at every step without losing previous outputs.
2
Build and compile StateGraph
↓
def build_pipeline(): graph = StateGraph(ResumeState) graph.add_node("jd_parser", jd_parser_node) graph.add_node("evaluator", evaluator_node) graph.add_node("gap_analyst", gap_analyst_node) graph.add_node("writer", writer_node) graph.set_entry_point("jd_parser") graph.add_edge("jd_parser", "evaluator") graph.add_edge("evaluator", "gap_analyst") graph.add_edge("gap_analyst", "writer") graph.add_edge("writer", END) return graph.compile() def run_pipeline(resume_text, jd_text, include_suggested_projects=True) -> ResumeState: pipeline = build_pipeline() return pipeline.invoke({ "resume_text": resume_text, "jd_text": jd_text, "include_suggested_projects": include_suggested_projects, "parsed_jd": None, "evaluation": None, "gap_analysis": None, "final_resume": None, "current_step": "starting", "error": None })
